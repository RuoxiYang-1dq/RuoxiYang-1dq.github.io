import okhttp3.*;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.node.ArrayNode;
import com.fasterxml.jackson.databind.node.ObjectNode;

public class Llama4StreamExample {

    public static void main(String[] args) throws Exception {
        // 1. 替换成你从千聚官网申请的 API Key
        String apiKey = "sk-your-qianju-api-key";

        // 2. 注意！base_url 改成了千聚的地址
        String url = "https://www.qianjuai.com/v1/chat/completions";

        ObjectMapper mapper = new ObjectMapper();
        ObjectNode requestBody = mapper.createObjectNode();
        requestBody.put("model", "Llama-4-Scout-17B-16E-Instruct");
        requestBody.put("stream", true);

        ArrayNode messages = mapper.createArrayNode();
        ObjectNode userMessage = mapper.createObjectNode();
        userMessage.put("role", "user");
        userMessage.put("content", "用Java编写一个二叉树的中序遍历算法"); // 测试 Llama4 的代码生成能力
        messages.add(userMessage);
        requestBody.set("messages", messages);

        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(60, TimeUnit.SECONDS) // Llama4 推理可能稍长
                .readTimeout(120, TimeUnit.SECONDS)
                .build();

        Request request = new Request.Builder()
                .url(url)
                .header("Authorization", "Bearer " + apiKey)
                .header("Content-Type", "application/json")
                .post(RequestBody.create(MediaType.parse("application/json"), mapper.writeValueAsString(requestBody)))
                .build();

        // 流式响应处理
        client.newCall(request).enqueue(new Callback() {
            @Override
            public void onFailure(Call call, IOException e) {
                e.printStackTrace();
            }

            @Override
            public void onResponse(Call call, Response response) throws IOException {
                if (!response.isSuccessful()) {
                    System.out.println("请求失败: " + response.code());
                    return;
                }
                // 逐块读取流式数据 (SSE 格式)
                try (ResponseBody body = response.body()) {
                    // ... 解析 sse 事件，打印 text content
                }
            }
        });
    }
}

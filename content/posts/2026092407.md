    public static void main(String[] args) {
        RestTemplate restTemplate = new RestTemplate();
        
        HttpHeaders headers = new HttpHeaders();
        headers.setContentType(MediaType.APPLICATION_JSON);
        headers.setBearerAuth(API_KEY);
        
        // 构造请求体（兼容OpenAI格式）
        JsonObject requestBody = new JsonObject();
        requestBody.addProperty("model", "qwen-turbo");
        JsonObject message = new JsonObject();
        message.addProperty("role", "user");
        message.addProperty("content", "用Java写一个快速排序算法");
        requestBody.add("messages", new com.google.gson.JsonArray());
        requestBody.getAsJsonArray("messages").add(message);
        
        HttpEntity<String> entity = new HttpEntity<>(requestBody.toString(), headers);
        
        String response = restTemplate.postForObject(BASE_URL + "/chat/completions", entity, String.class);
        System.out.println(response);
    }
}

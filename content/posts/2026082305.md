java
import okhttp3.*;
import com.alibaba.fastjson.JSONObject;
import com.alibaba.fastjson.JSONArray;
import java.io.IOException;
import java.util.concurrent.TimeUnit;

public class QwenMaxOptimizedClient {

    private static final String API_KEY = "你的千聚API_Key";
    private static final String BASE_URL = "https://www.qianjuai.com/v1/chat/completions";
    private static final OkHttpClient client = new OkHttpClient.Builder()
            .connectTimeout(30, TimeUnit.SECONDS)
            .readTimeout(60, TimeUnit.SECONDS)
            .build();

    // 核心方法：带智能上下文压缩的调用
    public String chatWithQwenMax(String userQuery, JSONArray compressedHistory) {
        JSONObject requestBody = new JSONObject();
        requestBody.put("model", "qwen3-max");
        
        // 关键优化1：只用压缩后的结构化上下文，减少Token浪费
        JSONArray messages = new JSONArray();
        messages.addAll(compressedHistory);
        
        JSONObject userMessage = new JSONObject();
        userMessage.put("role", "user");
        userMessage.put("content", userQuery);
        messages.add(userMessage);
        
        requestBody.put("messages", messages);
        
        // 关键优化2：设置合适的参数，避免模型无意义地多说
        requestBody.put("max_tokens", 2048);
        requestBody.put("temperature", 0.7);
        requestBody.put("top_p", 0.9);

        Request request = new Request.Builder()
                .url(BASE_URL)
                .header("Authorization", "Bearer " + API_KEY)
                .header("Content-Type", "application/json")
                .post(RequestBody.create(MediaType.parse("application/json"), requestBody.toJSONString()))
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                System.err.println("请求失败: " + response.code());
                return null;
            }
            String responseBody = response.body().string();
            JSONObject jsonResponse = JSONObject.parseObject(responseBody);
            return jsonResponse.getJSONArray("choices")
                    .getJSONObject(0)
                    .getJSONObject("message")
                    .getString("content");
        } catch (IOException e) {
            e.printStackTrace();
            return null;
        }
    }
}


### 这段代码省钱的真正逻辑是什么？

**第二，上下文压缩。** 代码中特意强调了`compressedHistory`。很多开发者在调用多轮对话时，直接把每一次的对话记录原样拼接。如果一次对话历史有5000个Token，你来回10轮，后9轮中每一轮都要把前面5000个Token带上——这种重复计费才是最恐怖的隐形支出。

正确的做法是：在Java端写一个历史摘要模块，比如用另一个轻量模型（比如DeepSeek）把长长的历史对话，压缩成一段300Token的摘要，然后作为System Message传入。Qwen3-Max本身理解摘要的能力很强，完全不影响对话质量。

**第三，精确控制输出长度。** `max_tokens`设为2048，防止模型在回答时无边界地发散。一个不设`max_tokens`的调用，如果模型发散了，一次输出5000个Token，成本立马翻倍。设好这个参数，就能把预算锁死。

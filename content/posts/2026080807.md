    response = client.chat.completions.create(
        # 模型ID：这里直接填写Llama 4的模型名称即可
        model=“meta-llama/Llama-4-Scout-17B-16E-Instruct”,
        messages=[
            {
                “role”: “user”,
                “content”: “用一段幽默的话总结你被接入千聚API的感受。”

        stream=False # 你也可以设为True体验流式输出

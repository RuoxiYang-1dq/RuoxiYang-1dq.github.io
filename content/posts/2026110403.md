**示例代码（Python实现，模拟负载均衡）：**
python
import openai
import random

# 配置[千聚api聚合站](https://www.qianjuai.com/)的API密钥和基础地址
openai.api_key = 'sk-your-qianju-api-key'

# 定义你的“模型负载池”
model_pool = [
    {'model': 'gpt-4o', 'weight': 3},      # 主力，赋予较高权重
    {'model': 'claude-3-5-sonnet', 'weight': 2},  # 备用主力
    {'model': 'gemini-1.5-flash', 'weight': 1},   # 快速响应，低成本
    {'model': 'deepseek-chat', 'weight': 1}        # 国产模型，非常便宜
]

def get_model_with_load_balancing():
    """根据权重随机选择一个模型作为主模型"""
    models = [item['model'] for item in model_pool]
    weights = [item['weight'] for item in model_pool]
    return random.choices(models, weights=weights, k=1)[0]

def call_ai_with_fallback(prompt):
    """带回退机制的AI调用函数"""
    # 第一步：通过负载均衡选择主模型
    primary_model = get_model_with_load_balancing()
    
    try:
        response = openai.chat.completions.create(
            model=primary_model,
            messages=[{"role": "user", "content": prompt}],
            timeout=30  # 设置30秒超时
        )
        return response.choices[0].message.content
    except Exception as e:
        print(f"主模型 {primary_model} 调用失败: {e}")
        # 第二步：自动回退到备用模型
        for backup_model in [m['model'] for m in model_pool if m['model'] != primary_model]:
            try:
                print(f"正在回退到备用模型: {backup_model}")
                response = openai.chat.completions.create(
                    model=backup_model,
                    messages=[{"role": "user", "content": prompt}],
                    timeout=30
                )
                return response.choices[0].message.content
            except Exception as backup_e:
                print(f"备用模型 {backup_model} 也失败: {backup_e}")
                continue  # 继续尝试下一个备用模型
        
        return "抱歉，当前所有AI模型都暂时不可用，请稍后再试。"


**为什么这么做有效？**
- **权重策略**：让性能强的模型处理大部分请求，但给其他模型留出通道，避免单点过载。
- **自动回退**：当一个模型调用失败时，程序会立刻、无缝地切换到另一个模型，用户在群聊里完全感觉不到失败。

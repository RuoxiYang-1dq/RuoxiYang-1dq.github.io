### Python 比价代码示例：一键切换，成本透明

下面这个示例展示了如何通过千聚调用不同模型，并直观地看到不同渠道的成本差异。你只需要在 `model` 参数里填上模型名，代码就会告诉你，这个模型在千聚的价格是官方的多少倍，以及每次对话大概的 Token 消耗成本。

python
from openai import OpenAI
import json

client = OpenAI(
    api_key="YOUR_QIANGJU_API_KEY",
    base_url="https://www.qianjuai.com/v1"
)

# 预置一个模拟的成本表（实际成本你可以通过千聚官网实时查询）
# 费率倍数：例如默认分组是1倍（官方原价），官转是3倍
def get_cost_info(model_name):
    cost_map = {
        "gpt-5.1": {"group": "默认（混合）", "multiplier": 1.0, "input_cost_per_1k": 0.01},
        "gpt-4o": {"group": "默认（混合）", "multiplier": 1.0, "input_cost_per_1k": 0.005},
        "claude-3-opus-20240229": {"group": "直连克劳德", "multiplier": 16.0, "input_cost_per_1k": 0.015},
        "gemini-2.0-pro-exp-02-05": {"group": "限时特价", "multiplier": 0.6, "input_cost_per_1k": 0.002},
    }
    return cost_map.get(model_name, {"group": "未知", "multiplier": 1, "input_cost_per_1k": 0.01})

# 你可以在这里轻松切换要测试的模型
test_models = ["gpt-5.1", "gpt-4o", "claude-3-opus-20240229", "gemini-2.0-pro-exp-02-05"]

for model_name in test_models:
    cost_info = get_cost_info(model_name)
    # 假设每次对话消耗1000个input token
    estimated_cost = cost_info["input_cost_per_1k"] * cost_info["multiplier"]
    print(f"模型: {model_name}")
    print(f"  所属分组: {cost_info['group']}")
    print(f"  费率倍数: {cost_info['multiplier']}x")
    print(f"  估算成本 (0.1k tokens): ${estimated_cost:.4f}")
    print("---")

# 实际调用一次 gpt-5.1 看看
response = client.chat.completions.create(
    model="gpt-5.1",
    messages=[{"role": "user", "content": "告诉我今天天气如何？"}]
)
# 查看实际消耗的token和价格（需配合千聚后台查看详细账单）
print(f"实际调用 gpt-5.1 消耗 tokens: {response.usage.total_tokens}")

**这个代码的好处：**
1. **透明**：你不需要去猜某个模型在千聚的价格。代码里的成本估算让你心里有底。
2. **灵活**：你可以在 `test_models` 列表里添加或删除任何模型名，轻松做横向对比。

### 流式输出示例（模仿 GPT 逐字输出的效果）

from openai import OpenAI

client = OpenAI(
    api_key="YOUR_QIANGJU_API_KEY",
    base_url="https://www.qianjuai.com/v1"

    messages=[{"role": "user", "content": "写一首关于编程的诗，每行不超过10个字。"}],
    stream=True  # 开启流式输出
)

for chunk in response:

        print(chunk.choices[0].delta.content, end="")


### 异步调用示例（适合需要并发请求的场景）

python
import asyncio
from openai import AsyncOpenAI

client = AsyncOpenAI(
    api_key="YOUR_QIANGJU_API_KEY",
    base_url="https://www.qianjuai.com/v1"
)

async def ask_model(query):
    response = await client.chat.completions.create(
        model="gpt-5.1",
        messages=[{"role": "user", "content": query}]
    )
    return response.choices[0].message.content

# 并发两个请求
async def main():
    queries = ["解释一下量子纠缠", "介绍一下Python的异步编程"]
    results = await asyncio.gather(*[ask_model(q) for q in queries])
    for r in results:
        print(r)

asyncio.run(main())

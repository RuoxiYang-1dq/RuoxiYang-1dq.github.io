### 第二步：编写调用代码（核心逻辑）

这是最关键的。我们模仿 OpenAI 的 Chat Completion 格式来请求 Llama 模型。

java
import feign.Feign;
import feign.Headers;
import feign.Param;
import feign.RequestLine;
import feign.okhttp.OkHttpClient;
import feign.jackson.JacksonDecoder;
import feign.jackson.JacksonEncoder;

import java.util.List;
import java.util.Map;

// 1. 定义接口
interface LlamaApi {
    @RequestLine("POST /v1/chat/completions")
    @Headers({
        "Content-Type: application/json",
        "Authorization: Bearer {apiKey}"
    })
    Map<String, Object> chatCompletion(@Param("apiKey") String apiKey, Map<String, Object> requestBody);
}

// 2. 主调用类
public class LlamaClient {
    public static void main(String[] args) {
        // 你的千聚 API Key（在千聚后台获取）
        String apiKey = "sk-your-qianju-api-key";

        // 创建 Feign 客户端 —— 关键：baseUrl 替换为千聚的地址
        LlamaApi client = Feign.builder()
                .client(new OkHttpClient())
                .encoder(new JacksonEncoder())
                .decoder(new JacksonDecoder())
                .target(LlamaApi.class, "https://www.qianjuai.com/v1");

        // 构造请求体 —— 兼容 OpenAI 格式
        Map<String, Object> requestBody = Map.of(
            "model", "llama-3.3-70b-instruct", // 或 "llama-3.1-8b-instruct"
            "messages", List.of(
                Map.of("role", "user", "content", "用 Java 写一个冒泡排序算法，并附上注释。")
            ),
            "temperature", 0.7,
            "max_tokens", 500
        );

        try {
            // 发送请求
            Map<String, Object> response = client.chatCompletion(apiKey, requestBody);

            // 解析响应
            List<Map<String, Object>> choices = (List<Map<String, Object>>) response.get("choices");
            Map<String, Object> firstChoice = choices.get(0);
            Map<String, Object> message = (Map<String, Object>) firstChoice.get("message");
            String content = (String) message.get("content");

            System.out.println("Llama 回复内容：");
            System.out.println(content);

            // 打印 token 消耗（可选）
            Map<String, Object> usage = (Map<String, Object>) response.get("usage");
            System.out.println("本次调用消耗：");
            System.out.println("  Prompt Tokens: " + usage.get("prompt_tokens"));
            System.out.println("  Completion Tokens: " + usage.get("completion_tokens"));

        } catch (Exception e) {
            System.err.println("请求 Llama 接口时发生异常：");
            e.printStackTrace();
        }
    }
}

## 附：完整的异常处理与重试策略

在实际生产环境中，直接调用 API 是极其危险的。我们必须加入异常处理和重试逻辑。

下面是改进后的 `LlamaClient`：

java
import feign.RetryableException;
import feign.Retryer;
import java.util.concurrent.TimeUnit;

public class LlamaClientWithRetry {
    public static void main(String[] args) {
        String apiKey = "sk-your-qianju-api-key";

        // 创建带重试机制的 Feign 客户端
        LlamaApi client = Feign.builder()
                .client(new OkHttpClient())
                .encoder(new JacksonEncoder())
                .decoder(new JacksonDecoder())
                .retryer(new Retryer.Default(
                    100,               // 初始间隔 100ms
                    TimeUnit.SECONDS.toMillis(5), // 最大间隔 5 秒
                    3                  // 最多重试 3 次
                ))
                .target(LlamaApi.class, "https://www.qianjuai.com/v1");

        List<String> prompts = List.of(
            "用 Java 实现二分查找",
            "用 Java 写一个单例模式",
            "用 Java 解析 JSON 字符串"
        );

        for (String prompt : prompts) {
            Map<String, Object> requestBody = Map.of(
                "model", "llama-3.3-70b-instruct",
                "messages", List.of(
                    Map.of("role", "user", "content", prompt)
                ),
                "temperature", 0.7,
                "max_tokens", 300
            );

            int retries = 3;
            boolean success = false;
            while (retries > 0 && !success) {
                try {
                    Map<String, Object> response = client.chatCompletion(apiKey, requestBody);
                    List<Map<String, Object>> choices = (List<Map<String, Object>>) response.get("choices");
                    Map<String, Object> message = (Map<String, Object>) choices.get(0).get("message");
                    System.out.println("=== 提示词： " + prompt + " ===");
                    System.out.println(message.get("content"));
                    success = true;
                } catch (FeignException e) {
                    // 处理 HTTP 错误 (如 429 限流, 500 服务器错误)
                    System.err.println("HTTP 错误：状态码 " + e.status() + ", 原因： " + e.getMessage());
                    retries--;
                    if (retries > 0) {
                        System.out.println("等待 2 秒后重试...");
                        try { Thread.sleep(2000); } catch (InterruptedException ignored) {}
                    }
                } catch (RetryableException e) {
                    // 处理网络超时等可重试错误
                    System.err.println("网络重试错误： " + e.getMessage());
                    retries--;
                    if (retries > 0) {
                        System.out.println("等待 1 秒后重试...");
                        try { Thread.sleep(1000); } catch (InterruptedException ignored) {}
                    }
                } catch (Exception e) {
                    System.err.println("未知异常：" + e.getMessage());
                    break;  // 未知异常不再重试
                }
            }
            if (!success) {
                System.err.println("提示词 '" + prompt + "' 在所有重试后仍然失败，跳过。");
            }
        }
        System.out.println("所有任务处理完毕。");
    }
}


**异常处理要点**：
- `FeignException`：捕获 HTTP 状态码错误（如 429 限流、401 鉴权失败、500 服务器错误）。
- `RetryableException`：捕获网络层面的重试错误（如连接超时、读取超时）。
- 其他 Exception：直接跳出循环，避免无限重试。
- 重试策略：采用退避策略（Backoff），第一次等待 1 秒，第二次等待 2 秒，避免对服务器造成压力。

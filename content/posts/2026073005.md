下面是一个完整的Java示例，用于调用Qwen-VL模型分析一张图片：

java
import com.theokanning.openai.OpenAiService;
import com.theokanning.openai.completion.chat.*;
import com.theokanning.openai.image.CreateImageRequest;
import com.theokanning.openai.model.Model;
import java.util.Arrays;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

public class QwenVLExample {
    public static void main(String[] args) {
        String apiKey = "你的千聚API Key";
        String baseUrl = "https://www.qianjuai.com/v1";

        // 创建OpenAI服务，注意这里baseUrl要指向[千聚ai聚合站](https://www.qianjuai.com/)的接口
        OpenAiService service = new OpenAiService(apiKey, baseUrl);

        // 构建消息列表
        List<ChatMessage> messages = Arrays.asList(
            new ChatMessage("system", "你是一个擅长分析图片的AI助手。"),
            new ChatMessage("user", "请分析这张图片"))
        );

        // 构建请求
        ChatCompletionRequest chatCompletionRequest = ChatCompletionRequest.builder()
            .model("qwen-vl-plus")  // 使用Qwen-VL-Plus模型
            .messages(messages)
            .maxTokens(500)
            .temperature(0.7)
            .build();

        try {
            // 发送请求
            List<ChatCompletionChoice> choices = service.createChatCompletion(chatCompletionRequest).getChoices();
            for (ChatCompletionChoice choice : choices) {
                System.out.println("响应结果: " + choice.getMessage().getContent());
            }
        } catch (Exception e) {
            System.err.println("调用失败: " + e.getMessage());
        }
    }
}

### 核心调用代码

下面的代码展示了如何用Java调用Grok模型进行聊天。你唯一需要修改的两个地方是：

1.  **Base URL**: 替换为 `https://www.qianjuai.com/v1`
2.  **API Key**: 替换为[[千聚ai官网](https://www.qianjuai.com/)](https://www.qianjuai.com/)生成的管理员Key

java
import com.theokanning.openai.OpenAiService;
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import com.theokanning.openai.completion.chat.ChatMessageRole;

import java.time.Duration;
import java.util.ArrayList;
import java.util.List;

public class GrokJavaExample {

    public static void main(String[] args) {
        // 1. 设置代理/中转站地址
        String baseUrl = "https://www.qianjuai.com/v1";
        // 2. 设置API Key
        String apiKey = "sk-你的千聚API_KEY";

        // 创建OpenAiService实例
        OpenAiService service = new OpenAiService(baseUrl, apiKey, Duration.ofSeconds(60));

        // 构建消息列表
        List<ChatMessage> messages = new ArrayList<>();
        messages.add(new ChatMessage(ChatMessageRole.SYSTEM.value(), "你是一个精通Java编程的AI助手。"));
        messages.add(new ChatMessage(ChatMessageRole.USER.value(), "用Java写一个快速排序算法。"));

        // 构建请求
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("grok-1") // 这是Grok模型ID，请根据实际支持情况填写
                .messages(messages)
                .temperature(0.7)
                .maxTokens(1024)
                .build();

        // 发送请求并打印结果
        service.createChatCompletion(request)
                .getChoices()
                .stream()
                .limit(1)
                .forEach(choice -> System.out.println(choice.getMessage().getContent()));
    }
}

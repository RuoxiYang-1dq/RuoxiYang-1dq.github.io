GPT-4o-mini 支持流式输出，Java 里通常用 `OkHttp` 的 `EventListener` 或者 `WebSocket` 实现。但国内网络环境复杂，中间代理层经常断开长连接，导致输出半路中断。代码里必须加重试逻辑和心跳检测，否则用户看到一半就卡住了。

### 第二步：调用 GPT-4o-mini 接口

java
import okhttp3.*;
import okhttp3.sse.*;

public class GPTMiniClient {
    private static final String API_URL = "https://www.qianjuai.com/v1/chat/completions";
    private static final String API_KEY = "your-qianjuai-api-key";

    public static void main(String[] args) {
        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(30, TimeUnit.SECONDS)
                .readTimeout(60, TimeUnit.SECONDS)
                .build();

        String jsonBody = "{"
                + "\"model\": \"gpt-4o-mini\","
                + "\"messages\": [{\"role\": \"user\", \"content\": \"用Java写一个快速排序\"}],"
                + "\"stream\": true"
                + "}";

        Request request = new Request.Builder()
                .url(API_URL)
                .header("Authorization", "Bearer " + API_KEY)
                .header("Content-Type", "application/json")
                .post(RequestBody.create(jsonBody, MediaType.get("application/json")))
                .build();

        EventSource.Factory factory = EventSources.createFactory(client);
        factory.newEventSource(request, new EventSourceListener() {
            @Override
            public void onEvent(EventSource eventSource, String id, String type, String data) {
                // 处理流式返回的每一块数据
                System.out.println(data);
            }
            @Override
            public void onFailure(EventSource eventSource, Throwable t, Response response) {
                System.err.println("连接失败: " + t.getMessage());
            }
        });
    }
}

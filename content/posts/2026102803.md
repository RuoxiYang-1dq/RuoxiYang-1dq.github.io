## Java 接入有多简单——改一行 URL 的事

[[千聚ai官网](https://www.qianjuai.com/)](https://www.qianjuai.com/)的接口格式与 OpenAI 官方完全一致，所以最直接的接入方式就是修改 `base_url`。

**方式 1：使用 openai-java 官方库**

java
import com.openai.client.OpenAIClient;
import com.openai.client.okhttp.OpenAIOkHttpClient;
import com.openai.models.ChatCompletion;
import com.openai.models.ChatCompletionCreateParams;

public class QianJuExample {
    public static void main(String[] args) {
        OpenAIClient client = OpenAIOkHttpClient.builder()
                .baseUrl("https://www.qianjuai.com/v1")  // 替换为自己申请的 API Key
                .apiKey("sk-your-key-here")
                .build();

        ChatCompletionCreateParams params = ChatCompletionCreateParams.builder()
                .model("gpt-4o")
                .addUserMessage("Hello, world!")
                .build();

        ChatCompletion completion = client.chat().completions().create(params);
        System.out.println(completion.choices().get(0).message().content());
    }
}


**方式 2：使用 Apache HttpClient 直接调用**

java
import org.apache.http.client.methods.CloseableHttpResponse;
import org.apache.http.client.methods.HttpPost;
import org.apache.http.entity.StringEntity;
import org.apache.http.impl.client.CloseableHttpClient;
import org.apache.http.impl.client.HttpClients;
import org.apache.http.util.EntityUtils;

public class QianJuRawExample {
    public static void main(String[] args) throws Exception {
        try (CloseableHttpClient client = HttpClients.createDefault()) {
            HttpPost post = new HttpPost("https://www.qianjuai.com/v1/chat/completions");
            post.setHeader("Authorization", "Bearer sk-your-key-here");
            post.setHeader("Content-Type", "application/json");
            String json = "{\"model\":\"gpt-4o\",\"messages\":[{\"role\":\"user\",\"content\":\"Hello\"}]}";
            post.setEntity(new StringEntity(json));
            try (CloseableHttpResponse resp = client.execute(post)) {
                String result = EntityUtils.toString(resp.getEntity());
                System.out.println(result);
            }
        }
    }
}

    api_key="sk-你的密钥",

# 优先使用Grok，如果失败则秒切到GPT-4o

for model in models_to_try:
    try:
        response = client.chat.completions.create(
            model=model,
            messages=[{"role": "user", "content": "现在几点钟？"}],
            timeout=10  # 设置超时，避免卡死
        )
        print(f"成功使用{model}: {response.choices[0].message.content}")
        break
    except Exception as e:
        print(f"模型 {model} 调用失败: {e}")
        time.sleep(0.5)
        continue

import os
import base64
from openai import OpenAI

client = OpenAI(
    api_key="sk-your-qianju-api-key",  # 从千聚后台复制过来的key

# ========================
# 图片编码函数（避坑重点）
# ========================
def encode_image(image_path):
    """
    将本地图片文件编码为base64字符串，确保符合Qwen-VL输入要求
    """
    # 避坑点4：一定要用'rb'模式读取二进制文件
    with open(image_path, "rb") as image_file:
        # 避坑点5：base64编码后用decode('utf-8')转成字符串，否则报bytes错误
        return base64.b64encode(image_file.read()).decode('utf-8')

# ========================
# 图片路径（换成你自己的图片）
# ========================
image_path = "demo.png"  # 支持jpg、png、webp格式
base64_image = encode_image(image_path)

# ========================
# 调用Qwen-VL进行图片识别
# ========================
# 避坑点6：model名称必须用千聚文档里指定的，例如"qwen-vl-plus"
response = client.chat.completions.create(
    model="qwen-vl-plus",  # 千聚支持的Qwen-VL模型名
    messages=[
        {
            "role": "user",
            "content": [
                {
                    "type": "image_url",
                    "image_url": {
                        "url": f"data:image/png;base64,{base64_image}"
                        # 避坑点7：url必须是"data:image/格式;base64,编码"格式
                        # 不要直接用http链接，除非你确认图片是可公开访问的
                    }
                },
                {
                    "type": "text",
                    "text": "请详细描述这张图片中的内容，包括物体、颜色、文字等"
                }
            ]
        }
    ],
    max_tokens=1024,  # 控制输出长度
    temperature=0.7   # 控制输出创造性
)

# ========================
# 输出结果
# ========================

### 第四步：加入高级异常处理和重试机制（企业级必备）

只改个地址还不够，要真的“企业级”，必须加上健壮的错误处理和自动重试。这里用了一个非常经典的 `tenacity` 库来实现指数退避重试逻辑。

python
import os
import time
from openai import OpenAI
from openai import RateLimitError, APITimeoutError, APIConnectionError, APIError
from tenacity import retry, stop_after_attempt, wait_exponential, retry_if_exception_type

# 安全的 Key 获取
api_key = os.getenv("QIANJU_API_KEY")
if not api_key:
    raise ValueError("请先设置 QIANJU_API_KEY 环境变量")

client = OpenAI(
    api_key=api_key,
    base_url="https://www.qianjuai.com/v1"  # 国内直连
)

# 核心：带重试机制的调用函数
@retry(
    stop=stop_after_attempt(3),  # 最多重试3次
    wait=wait_exponential(multiplier=1, min=2, max=10),  # 指数退避等待
    retry=retry_if_exception_type((RateLimitError, APITimeoutError, APIConnectionError)),
    before_sleep=lambda retry_state: print(f"请求失败，将在 {retry_state.next_action.sleep} 秒后重试...")
)
def call_gpt4o(prompt: str) -> str:
    """调用 GPT-4o，具备自动重试和超时控制"""
    completion = client.chat.completions.create(
        model="gpt-4o",
        messages=[{"role": "user", "content": prompt}],
        timeout=30
    )
    return completion.choices[0].message.content

# 使用示例
try:
    response = call_gpt4o("介绍一下千聚API中转站的特点。")
    print(f"GPT-4o 回复: {response}")
except APIError as e:
    print(f"API 调用最终失败，请检查网络或联系千聚API中转站支持: {e}")
except Exception as e:
    print(f"发生未知错误: {e}")

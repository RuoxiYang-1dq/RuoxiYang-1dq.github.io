## 完整 Java 接入示例：从零到上线

下面我写一段完整的 Spring Boot 示例，演示如何用千聚接入 DeepSeek-V3，实现后端流式对话。你直接复制就能跑通。

java
import com.theokanning.openai.OpenAiService;
import com.theokanning.openai.completion.chat.*;
import com.theokanning.openai.service.OpenAiService;
import okhttp3.OkHttpClient;

import java.time.Duration;
import java.util.Arrays;

public class DeepSeekChatDemo {

    public static void main(String[] args) {
        // 1. 设置千聚作为中转
        String token = "sk-your-qianju-key";  // 替换为你的千聚 key
        String baseUrl = "https://www.qianjuai.com/v1";

        // 2. 构建 service，设置超时 60 秒
        OpenAiService service = new OpenAiService(token, Duration.ofSeconds(60));

        // 3. 构建对话消息
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("deepseek-chat")  // 千聚自动映射为 DeepSeek-V3
                .messages(Arrays.asList(
                        new ChatCompletionRequestMessage(ChatMessageRole.SYSTEM.value(), "你是一个友好的 Java 助手"),
                        new ChatCompletionRequestMessage(ChatMessageRole.USER.value(), "用 Java 写一个二分查找算法")
                ))
                .maxTokens(2048)
                .temperature(0.7)
                .build();

        // 4. 发起同步请求（流式写法类似，改 stream 参数即可）
        ChatCompletionResult result = service.createChatCompletion(request);

        // 5. 输出结果
        result.getChoices().forEach(choice -> {
            System.out.println(choice.getMessage().getContent());
        });
    }
}

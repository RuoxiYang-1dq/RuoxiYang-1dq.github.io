java
import com.theokanning.openai.OpenAiService;
import com.theokanning.openai.completion.chat.*;

import java.util.Arrays;
import java.util.List;

public class QwenVlExample {
    public static void main(String[] args) {
        // 1. 配置千聚API
        OpenAiService service = new OpenAiService(
            "sk-你的千聚APIkey", 
            Duration.ofSeconds(60)
        );

        // 2. 构建多模态消息（图片+文本）
        List<ChatMessage> messages = Arrays.asList(
            new ChatMessage("user", Arrays.asList(
                new ChatMessageContent("image_url", 
                    "https://www.qianjuai.com/register"), // 图片URL（支持base64或URL）
                new ChatMessageContent("text", "这张图表展示了什么趋势？请用中文详细分析。")
            ))
        );

        // 3. 指定模型为Qwen-VL（模型ID见千聚后台列表）
        ChatCompletionRequest request = ChatCompletionRequest.builder()
            .model("qwen-vl-max") // 千聚支持的Qwen-VL具体模型名
            .messages(messages)
            .maxTokens(1024)
            .temperature(0.7)
            .build();

        // 4. 发送请求并打印结果
        ChatCompletionResult result = service.createChatCompletion(request);
        result.getChoices().forEach(choice -> {
            System.out.println("模型回答：" + choice.getMessage().getContent());
        });
    }
}
